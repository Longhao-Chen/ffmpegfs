257 static void init_stat(struct stat * stbuf, size_t fsize, time_t ftime, bool directory)
258 {
259  memset(stbuf, 0, sizeof(struct stat));
260 
261  stbuf->st_mode = DEFFILEMODE; //S_IFREG | S_IRUSR | S_IRGRP | S_IROTH;
262  if (directory)
263  {
264  stbuf->st_mode |= S_IFDIR | S_IXUSR | S_IXGRP | S_IXOTH;
265  stbuf->st_nlink = 2;
266  }
267  else
268  {
269  stbuf->st_mode |= S_IFREG;
270  stbuf->st_nlink = 1;
271  }
272 
273 #if defined __x86_64__ || !defined __USE_FILE_OFFSET64
274  stbuf->st_size = static_cast<__off_t>(fsize);
275 #else
276  stbuf->st_size = static_cast<__off64_t>(fsize);
277 #endif
278  stbuf->st_blocks = (stbuf->st_size + 512 - 1) / 512;
279 
280  // Set current user as owner
281  stbuf->st_uid = getuid();
282  stbuf->st_gid = getgid();
283 
284  // Use current date/time
285  stbuf->st_atime = stbuf->st_mtime = stbuf->st_ctime = ftime;

317 static void prepare_script()
318 {
319  std::string scriptsource;
320 
321  exepath(&scriptsource);
322  scriptsource += params.m_scriptsource;
323 
324  Logging::debug(scriptsource, "Reading virtual script source.");
325 
326  FILE *fpi = fopen(scriptsource.c_str(), "rt");
327  if (fpi == nullptr)
328  {
329  Logging::warning(scriptsource, "File open failed. Disabling script: (%1) %2", errno, strerror(errno));
330  params.m_enablescript = false;
331  }
332  else
333  {
334  struct stat stbuf;
335  if (fstat(fileno(fpi), &stbuf) == -1)
336  {
337  Logging::warning(scriptsource, "File could not be accessed. Disabling script: (%1) %2", errno, strerror(errno));
338  params.m_enablescript = false;
339  }
340  else
341  {
342  script_file.resize(static_cast<size_t>(stbuf.st_size));
343 
344  if (fread(&script_file[0], 1, static_cast<size_t>(stbuf.st_size), fpi) != static_cast<size_t>(stbuf.st_size))
345  {
346  Logging::warning(scriptsource, "File could not be read. Disabling script: (%1) %2", errno, strerror(errno));
347  params.m_enablescript = false;
348  }
349  else
350  {
351  Logging::trace(scriptsource, "Read %1 bytes of script file.", script_file.size());
352  }
353  }
354 
355  fclose(fpi);
356  }

561 LPVIRTUALFILE find_original(std::string * filepath)
562 {
563  sanitise_filepath(filepath);
564 
565  LPVIRTUALFILE virtualfile = find_file(*filepath);
566 
567  errno = 0;
568 
569  if (virtualfile != nullptr)
570  {
571  *filepath = virtualfile->m_origfile;
572  return virtualfile;
573  }
574  else
575  {
576  // Fallback to old method (required if file accessed directly)
577  std::string ext;
578  if (!params.m_format[0].is_hls() && find_ext(&ext, *filepath) && (strcasecmp(ext, params.m_format[0].fileext()) == 0 || (params.smart_transcode() && strcasecmp(ext, params.m_format[1].fileext()) == 0)))
579  {
580  std::string dir(*filepath);
581  std::string filename(*filepath);
582  std::string tmppath;
583  struct dirent **namelist;
584  struct stat stbuf;
585  int count;
586  int found = 0;
587 
588  remove_filename(&dir);
589  tmppath = dir;
590 
591  count = scandir(dir.c_str(), &namelist, selector, nullptr);
592  if (count == -1)
593  {
594  if (errno != ENOTDIR) // If not a directory, simply ignore error
595  {
596  Logging::error(dir, "Error scanning directory: (%1) %2", errno, strerror(errno));
597  }
598  return nullptr;
599  }
600 
601  remove_path(&filename);
602  std::regex specialChars { R"([-[\]{}()*+?.,\^$|#\s])" };
603  filename = std::regex_replace(filename, specialChars, R"(\$&)");
604  replace_ext(&filename, "*");
605 
606  for (int n = 0; n < count; n++)
607  {
608  if (!found && !compare(namelist[n]->d_name, filename))
609  {
610  append_filename(&tmppath, namelist[n]->d_name);
611  found = 1;
612  }
613  free(namelist[n]);
614  }
615  free(namelist);
616 
617  sanitise_filepath(&tmppath);
618 
619  if (found && lstat(tmppath.c_str(), &stbuf) == 0)
620  {
621  // File exists with this extension
622  LPVIRTUALFILE virtualfile;
623 
624  if (*filepath != tmppath)
625  {
626  virtualfile = insert_file(VIRTUALTYPE_DISK, *filepath, tmppath, &stbuf);
627  *filepath = tmppath;
628  }
629  else
630  {
631  virtualfile = insert_file(VIRTUALTYPE_DISK, tmppath, &stbuf, VIRTUALFLAG_PASSTHROUGH);
632  }
633  return virtualfile;
634  }
635  else
636  {
637  // File does not exist; not an error
638  errno = 0;
639  }
640  }
641  }
642  // Source file exists with no supported extension, keep path
643  return nullptr;
644 }
