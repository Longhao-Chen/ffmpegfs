249 static void init_stat(struct stat * stbuf, size_t fsize, bool directory)
250 {
251  memset(stbuf, 0, sizeof(struct stat));
252 
253  stbuf->st_mode = DEFFILEMODE; //S_IFREG | S_IRUSR | S_IRGRP | S_IROTH;
254  if (directory)
255  {
256  stbuf->st_mode |= S_IFDIR | S_IXUSR | S_IXGRP | S_IXOTH;
257  stbuf->st_nlink = 2;
258  }
259  else
260  {
261  stbuf->st_mode |= S_IFREG;
262  stbuf->st_nlink = 1;
263  }
264 
265 #if defined __x86_64__ || !defined __USE_FILE_OFFSET64
266  stbuf->st_size = static_cast<__off_t>(fsize);
267 #else
268  stbuf->st_size = static_cast<__off64_t>(fsize);
269 #endif
270  stbuf->st_blocks = (stbuf->st_size + 512 - 1) / 512;
271 
272  // Set current user as owner
273  stbuf->st_uid = getuid();
274  stbuf->st_gid = getgid();
275 
276  // Use current date/time
277  stbuf->st_atime = stbuf->st_mtime = stbuf->st_ctime = time(nullptr);
278 }

283 static void prepare_script()
284 {
285  std::string scriptsource;
286 
287  exepath(&scriptsource);
288  scriptsource += params.m_scriptsource;
289 
290  Logging::debug(scriptsource, "Reading virtual script source.");
291 
292  FILE *fpi = fopen(scriptsource.c_str(), "rt");
293  if (fpi == nullptr)
294  {
295  Logging::warning(scriptsource, "File open failed. Disabling script: (%1) %2", errno, strerror(errno));
296  params.m_enablescript = false;
297  }
298  else
299  {
300  struct stat stbuf;
301  if (fstat(fileno(fpi), &stbuf) == -1)
302  {
303  Logging::warning(scriptsource, "File could not be accessed. Disabling script: (%1) %2", errno, strerror(errno));
304  params.m_enablescript = false;
305  }
306  else
307  {
308  script_file.resize(static_cast<size_t>(stbuf.st_size));
309 
310  if (fread(&script_file[0], 1, static_cast<size_t>(stbuf.st_size), fpi) != static_cast<size_t>(stbuf.st_size))
311  {
312  Logging::warning(scriptsource, "File could not be read. Disabling script: (%1) %2", errno, strerror(errno));
313  params.m_enablescript = false;
314  }
315  else
316  {
317  Logging::trace(scriptsource, "Read %1 bytes of script file.", script_file.size());
318  }
319  }
320 
321  fclose(fpi);
322  }
323 }

347 static bool transcoded_name(std::string * filepath, FFmpegfs_Format **current_format /*= nullptr*/)
348 {
349  AVOutputFormat* format = av_guess_format(nullptr, filepath->c_str(), nullptr);
350 
351  if (format != nullptr)
352  {
353  std::string ext;
354 
355  if (!find_ext(&ext, *filepath) || !std::binary_search(passthrough_map.begin(), passthrough_map.end(), ext, nocasecompare))
356  {
357  FFmpegfs_Format *ffmpegfs_format = params.current_format(*filepath);
358 
359  if ((ffmpegfs_format->audio_codec_id() != AV_CODEC_ID_NONE && format->audio_codec != AV_CODEC_ID_NONE) ||
360  (ffmpegfs_format->video_codec_id() != AV_CODEC_ID_NONE && format->video_codec != AV_CODEC_ID_NONE))
361  {
362  *current_format = params.current_format(*filepath);
363  replace_ext(filepath, (*current_format)->fileext());
364  return true;
365  }
366  }
367  }
368 
369  if (current_format != nullptr)
370  {
371  *current_format = nullptr;
372  }
373  return false;
374 }

454 int load_path(const std::string & path, const struct stat *statbuf, void *buf, fuse_fill_dir_t filler)
455 {
456  int title_count = 0;
457 
458  filenamemap::const_iterator it = filenames.lower_bound(path);
459  while (it != filenames.end())
460  {
461  const std::string & key = it->first;
462  if (key.compare(0, path.size(), path) == 0) // Really a prefix?
463  {
464  LPCVIRTUALFILE virtualfile = &it->second;
465  struct stat stbuf;
466  std::string destfile;
467 
468  get_destname(&destfile, virtualfile->m_origfile);
469  remove_path(&destfile);
470 
471  title_count++;
472 
473  memcpy(&stbuf, statbuf, sizeof(struct stat));
474 
475  stbuf.st_size = virtualfile->m_st.st_size;
476  stbuf.st_blocks = (stbuf.st_size + 512 - 1) / 512;
477 
478  if (buf != nullptr && filler(buf, destfile.c_str(), &stbuf, 0))
479  {
480  // break;
481  }
482  }
483  it++;
484  }
485 
486  return title_count;
487 }

515 LPVIRTUALFILE find_original(std::string * filepath)
516 {
517  sanitise_filepath(filepath);
518 
519  LPVIRTUALFILE virtualfile = find_file(*filepath);
520 
521  errno = 0;
522 
523  if (virtualfile != nullptr)
524  {
525  *filepath = virtualfile->m_origfile;
526  return virtualfile;
527  }
528  else
529  {
530  // Fallback to old method (required if file accessed directly)
531  std::string ext;
532  if (find_ext(&ext, *filepath) && (strcasecmp(ext, params.m_format[0].fileext()) == 0 || (params.smart_transcode() && strcasecmp(ext, params.m_format[1].fileext()) == 0)))
533  {
534  std::string dir(*filepath);
535  std::string filename(*filepath);
536  std::string tmppath;
537  struct dirent **namelist;
538  struct stat stbuf;
539  int count;
540  int found = 0;
541 
542  remove_filename(&dir);
543  tmppath = dir;
544 
545  count = scandir(dir.c_str(), &namelist, selector, nullptr);
546  if (count == -1)
547  {
548  if (errno != ENOTDIR) // If not a directory, simply ignore error
549  {
550  Logging::error(dir, "Error scanning directory: (%1) %2", errno, strerror(errno));
551  }
552  return nullptr;
553  }
554 
555  remove_path(&filename);
556  std::regex specialChars { R"([-[\]{}()*+?.,\^$|#\s])" };
557  filename = std::regex_replace(filename, specialChars, R"(\$&)");
558  replace_ext(&filename, "*");
559 
560  for (int n = 0; n < count; n++)
561  {
562  if (!found && !compare(namelist[n]->d_name, filename))
563  {
564  append_filename(&tmppath, namelist[n]->d_name);
565  found = 1;
566  }
567  free(namelist[n]);
568  }
569  free(namelist);
570 
571  sanitise_filepath(&tmppath);
572 
573  if (found && lstat(tmppath.c_str(), &stbuf) == 0)
574  {
575  // File exists with this extension
576  LPVIRTUALFILE virtualfile = insert_file(VIRTUALTYPE_DISK, *filepath, tmppath, &stbuf);
577  *filepath = tmppath;
578  return virtualfile;
579  }
580  else
581  {
582  // File does not exist; not an error
583  errno = 0;
584  }
585  }
586  }
587  // Source file exists with no supported extension, keep path
588  return nullptr;
589 }

643 static int ffmpegfs_readdir(const char *path, void *buf, fuse_fill_dir_t filler, off_t /*offset*/, struct fuse_file_info * /*fi*/)
644 {
645  std::string origpath;
646  DIR *dp;
647  struct dirent *de;
648 #if defined(USE_LIBBLURAY) || defined(USE_LIBDVD) || defined(USE_LIBVCD)
649  int res;
650 #endif
651 
652  Logging::trace(path, "readdir");
653 
654  translate_path(&origpath, path);
655  append_sep(&origpath);
656 
657  // Add a virtual script if enabled
658  if (params.m_enablescript)
659  {
660  std::string filename(params.m_scriptfile);
661  struct stat stbuf;
662 
663  init_stat(&stbuf, script_file.size(), false);
664 
665  if (filler(buf, filename.c_str(), &stbuf, 0))
666  {
667  // break;
668  }
669 
670  LPVIRTUALFILE virtualfile = insert_file(VIRTUALTYPE_SCRIPT, origpath + filename, &stbuf);
671 
672  virtualfile->m_file_contents = script_file;
673  }
674 
675 #ifdef USE_LIBVCD
676  res = check_vcd(origpath, buf, filler);
677  if (res != 0)
678  {
679  // Found VCD or error reading VCD
680  return (res >= 0 ? 0 : res);
681  }
682 #endif // USE_LIBVCD
683 #ifdef USE_LIBDVD
684  res = check_dvd(origpath, buf, filler);
685  if (res != 0)
686  {
687  // Found DVD or error reading DVD
688  return (res >= 0 ? 0 : res);
689  }
690 #endif // USE_LIBDVD
691 #ifdef USE_LIBBLURAY
692  res = check_bluray(origpath, buf, filler);
693  if (res != 0)
694  {
695  // Found Bluray or error reading Bluray
696  return (res >= 0 ? 0 : res);
697  }
698 #endif // USE_LIBBLURAY
699 
700  dp = opendir(origpath.c_str());
701  if (dp != nullptr)
702  {
