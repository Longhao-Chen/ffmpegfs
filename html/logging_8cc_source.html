81 Logging::Logger::~Logger()
82 {
83  if (!m_logging || m_loglevel > m_logging->m_max_level)
84  {
85  return;
86  }
87 
88  // Construct string containing time
89  time_t now = time(nullptr);
90  std::string time_string(30, '\0');
91  std::string loglevel;
92  std::string filename;
93  std::string msg;
94 
95  time_string.resize(strftime(&time_string[0], time_string.size(), "%F %T", localtime(&now))); // Mind the blank at the end
96 
97  loglevel = m_level_name_map.at(m_loglevel) + ":";
98 
99  msg = str();
100  rtrim(msg);
101 
102  if (!m_filename.empty())
103  {
104  filename = "[" + m_filename + "] ";
105  }
106 
107  if (m_logging->m_to_syslog)
108  {
109  syslog(m_syslog_level_map.at(m_loglevel), "%s %s%s", loglevel.c_str(), filename.c_str(), msg.c_str());
110  }
111 
112  if (m_logging->m_logfile.is_open())
113  {
114  m_logging->m_logfile << time_string << " " << loglevel << " " << filename << msg << std::endl;
115  }
116 
117  if (m_logging->m_to_stderr)
118  {
119  if (!filename.empty())
120  {
121  filename = COLOUR_LIGHT_PURPLE + filename + COLOUR_RESET;
122  }
123 
124  std::clog << COLOUR_DARK_GRAY << time_string << " " << loglevel << COLOUR_RESET << " " << filename << msg << std::endl;
125  }
126 }
