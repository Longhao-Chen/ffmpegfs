82 static bool transcode_until(Cache_Entry* cache_entry, size_t offset, size_t len, uint32_t segment_no)
83 {
84  size_t end = offset + len; // Cast OK: offset will never be < 0.
85  bool success = true;
86 
87  if (cache_entry->is_finished() || cache_entry->m_buffer->is_segment_finished(segment_no) || cache_entry->m_buffer->tell(segment_no) >= end)
88  {
89  return true;
90  }
91 
92  try
93  {
94  // Wait until decoder thread has reached the desired position
95  if (cache_entry->m_is_decoding)
96  {
97  bool reported = false;
98  while (!(cache_entry->is_finished() || cache_entry->m_buffer->is_segment_finished(segment_no) || cache_entry->m_buffer->tell(segment_no) >= end) && !cache_entry->m_cache_info.m_error)
99  {
100  if (fuse_interrupted())
101  {
102  Logging::info(cache_entry->destname(), "Client has gone away.");
103  throw false;
104  }
105 
106  if (thread_exit)
107  {
108  Logging::warning(cache_entry->destname(), "Received thread exit.");
109  throw false;
110  }
111 
112  if (!reported)
113  {
114  Logging::trace(cache_entry->destname(), "Segment no. %1: Cache miss at offset %2 (length %3), remaining %4.", segment_no, offset, len, format_size_ex(end - cache_entry->m_buffer->size(segment_no)).c_str());
115  reported = true;
116  }
117  sleep(0);
118  }
119 
120  if (reported)
121  {
122  Logging::trace(cache_entry->destname(), "Segment no. %1: Cache hit at offset %2 (length %3), remaining %4.", segment_no, offset, len, format_size_ex(end - cache_entry->m_buffer->size(segment_no)).c_str());
123  }
124  success = !cache_entry->m_cache_info.m_error;
125  }
126  }
127  catch (bool _success)
128  {
129  success = _success;
130  }
131 
132  return success;
133 }

141 static int transcode_finish(Cache_Entry* cache_entry, FFmpeg_Transcoder &transcoder)
142 {
143  int res = transcoder.encode_finish();
144  if (res < 0)
145  {
146  return res;
147  }
148 
149  // Check encoded buffer size. Does not affect HLS segments.
150  cache_entry->m_cache_info.m_duration = transcoder.duration();
151  cache_entry->m_cache_info.m_encoded_filesize = cache_entry->m_buffer->buffer_watermark();
152  cache_entry->m_cache_info.m_video_frame_count = transcoder.video_frame_count();
153  cache_entry->m_cache_info.m_segment_count = transcoder.segment_count();
154  cache_entry->m_cache_info.m_result = !transcoder.have_seeked() ? RESULTCODE_FINISHED_SUCCESS : RESULTCODE_FINISHED_INCOMPLETE;
155  cache_entry->m_is_decoding = false;
156  cache_entry->m_cache_info.m_errno = 0;
157  cache_entry->m_cache_info.m_averror = 0;
158 
159  Logging::debug(transcoder.destname(), "Finishing file.");
160 
161  if (!cache_entry->m_buffer->reserve(cache_entry->m_cache_info.m_encoded_filesize))
162  {
163  Logging::debug(transcoder.destname(), "Unable to truncate buffer.");
164  }
165 
166  if (!transcoder.is_multiformat())
167  {
168  Logging::debug(transcoder.destname(), "Predicted size: %1 Final: %2 Diff: %3 (%4%).",
169  format_size_ex(cache_entry->m_cache_info.m_predicted_filesize).c_str(),
170  format_size_ex(cache_entry->m_cache_info.m_encoded_filesize).c_str(),
171  format_result_size_ex(cache_entry->m_cache_info.m_encoded_filesize, cache_entry->m_cache_info.m_predicted_filesize).c_str(),
172  static_cast<double>((cache_entry->m_cache_info.m_encoded_filesize * 1000 / (cache_entry->m_cache_info.m_predicted_filesize + 1)) + 5) / 10);
173  }
174 
175  cache_entry->flush();
176 
177  return 0;
178 }

340 Cache_Entry* transcoder_new(LPVIRTUALFILE virtualfile, bool begin_transcode)
341 {
342  // Allocate transcoder structure
343  Cache_Entry* cache_entry = cache->open(virtualfile);
344  if (cache_entry == nullptr)
345  {
346  return nullptr;
347  }
348 
349  Logging::trace(cache_entry->filename(), "Creating transcoder object.");
350 
351  try
352  {
353  cache_entry->lock();
354 
355  if (!cache_entry->open(begin_transcode))
356  {
357  throw static_cast<int>(errno);
358  }
359 
360  if (params.m_disable_cache)
361  {
362  // Disable cache
363  cache_entry->clear();
364  }
365  else if (!cache_entry->m_is_decoding && cache_entry->outdated())
366  {
367  cache_entry->clear();
368  }
369 
370  if (cache_entry->m_cache_info.m_duration)
371  {
372  virtualfile->m_duration = cache_entry->m_cache_info.m_duration;
373  }
374  if (cache_entry->m_cache_info.m_predicted_filesize)
375  {
376  virtualfile->m_predicted_size = cache_entry->m_cache_info.m_predicted_filesize;
377  }
378  if (cache_entry->m_cache_info.m_video_frame_count)
379  {
380  virtualfile->m_video_frame_count = cache_entry->m_cache_info.m_video_frame_count;
381  }
382 
383  if (!cache_entry->m_is_decoding && !cache_entry->is_finished_success())
384  {
385  if (begin_transcode)
386  {
387  Logging::debug(cache_entry->filename(), "Starting transcoder thread.");
388 
389  // Clear cache to remove any older remains
390  cache_entry->clear();
391 
392  // Must decode the file, otherwise simply use cache
393  cache_entry->m_is_decoding = true;
394 
395  THREAD_DATA* thread_data = new(std::nothrow) THREAD_DATA;
396 
397  thread_data->m_initialised = false;
398  thread_data->m_arg = cache_entry;
399  thread_data->m_lock_guard = false;
400 
401  {
402  std::unique_lock<std::mutex> lock(thread_data->m_mutex);
403 
404  tp->schedule_thread(&transcoder_thread, thread_data);
405 
406  // Let decoder get into gear before returning from open
407  while (!thread_data->m_lock_guard)
408  {
409  thread_data->m_cond.wait(lock);
410  }
411  }
412 
413  if (cache_entry->m_cache_info.m_error)
414  {
415  int ret;
416 
417  Logging::trace(cache_entry->filename(), "Transcoder error!");
418 
419  ret = cache_entry->m_cache_info.m_errno;
420  if (!ret)
421  {
422  ret = EIO; // Must return something, be it a simple I/O error...
423  }
424  throw ret;
425  }
426 
427  Logging::debug(cache_entry->filename(), "Transcoder thread is running.");
428  }
429  else if (!cache_entry->m_cache_info.m_predicted_filesize)
430  {
431  if (!transcoder_predict_filesize(virtualfile, cache_entry))
432  {
433  throw static_cast<int>(errno);
434  }
435  }
436  }
437  else if (begin_transcode)
438  {
439  Logging::trace(cache_entry->destname(), "Reading file from cache.");
440  }
441 
442  cache_entry->unlock();
443  }
444  catch (int orgerrno)
445  {
446  cache_entry->m_is_decoding = false;
447  cache_entry->unlock();
448  cache->close(&cache_entry, CACHE_CLOSE_DELETE);
449  cache_entry = nullptr; // Make sure to return NULL here even if the cache could not be deleted now (still in use)
450  errno = orgerrno; // Restore last errno
451  }
452 
453  return cache_entry;
454 }

456 bool transcoder_read(Cache_Entry* cache_entry, char* buff, size_t offset, size_t len, int * bytes_read, uint32_t segment_no)
457 {
458  bool success = true;
459 
460  if (!segment_no)
461  {
462  Logging::trace(cache_entry->destname(), "Reading %1 bytes from offset %2.", len, offset);
463  }
464  else
465  {
466  Logging::trace(cache_entry->destname(), "Reading %1 bytes from offset %2 for segment no. %3.", len, offset, segment_no);
467  }
468 
469  // Store access time
470  cache_entry->update_access();
471 
472  // Update read counter
473  cache_entry->update_read_count();
474 
475  try
476  {
477  // Try to read requested segment, stack a seek to if if this fails.
478  // No seek if not HLS (segment_no) and not required if < MIN_SEGMENT
479  if (segment_no && !cache_entry->m_buffer->segment_exists(segment_no))
480  {
481  cache_entry->m_seek_to_no = segment_no;
482  }
483 
484  // Open for reading if necessary
485  if (!cache_entry->m_buffer->open_file(segment_no, CACHE_FLAG_RO))
486  {
487  throw false;
488  }
489 
490  if (!cache_entry->is_finished_success())
491  {
492  switch (params.current_format(cache_entry->virtualfile())->filetype())
493  {
494  case FILETYPE_MP3:
495  {
496  // If we are encoding to MP3 and the requested data overlaps the ID3v1 tag
497  // at the end of the file, do not encode data first up to that position.
498  // This optimises the case where applications read the end of the file
499  // first to read the ID3v1 tag.
500  if ((offset > cache_entry->m_buffer->tell(segment_no)) &&
501  (offset + len >= (cache_entry->size() - ID3V1_TAG_LENGTH)))
502  {
503  // Stuff buffer with garbage, apps won't try to play that chunk anyway.
504  memset(buff, 0xFF, len);
505  if (cache_entry->size() - offset == ID3V1_TAG_LENGTH)
506  {
507  memcpy(buff, &cache_entry->m_id3v1, std::min(len, ID3V1_TAG_LENGTH));
508  }
509 
510  errno = 0;
511 
512  throw true; // OK
513  }
514  break;
515  }
516  default:
517  {
518  break;
519  }
520  }
521 
522  // Windows seems to access the files on Samba drives starting at the last 64K segment simply when
523  // the file is opened. Setting win_smb_fix=1 will ignore these attempts (not decode the file up
524  // to this point).
525  // Access will only be ignored if occurring at the second access.
526  if (params.m_win_smb_fix && cache_entry->read_count() == 2)
527  {
528  if ((offset > cache_entry->m_buffer->tell(segment_no)) &&
529  (len <= 65536) &&
530  check_ignore(cache_entry->size(), offset) &&
531  ((offset + len) > (cache_entry->size())))
532  {
533  Logging::warning(cache_entry->destname(), "Ignoring Windows' groundless access to last 8K boundary of file.");
534 
535  errno = 0;
536  *bytes_read = 0; // We've read nothing
537  len = 0;
538 
539  throw true; // OK
540  }
541  }
542  }
543 
544  success = transcode_until(cache_entry, offset, len, segment_no);
545 
546  if (!success)
547  {
548  errno = cache_entry->m_cache_info.m_errno ? cache_entry->m_cache_info.m_errno : EIO;
549  throw false;
550  }
551 
552  // truncate if we didn't actually get len
553  if (cache_entry->m_buffer->buffer_watermark(segment_no) < offset)
554  {
555  len = 0;
556  }
557  else if (cache_entry->m_buffer->buffer_watermark(segment_no) < offset + len)
558  {
559  len = cache_entry->m_buffer->buffer_watermark(segment_no) - offset;
560  }
561 
562  if (!cache_entry->m_buffer->copy(reinterpret_cast<uint8_t*>(buff), offset, len, segment_no))
563  {
564  len = 0;
565  // We already capped len to not overread the buffer, so it is an error if we end here.
566  throw false;
567  }
568 
569  if (cache_entry->m_cache_info.m_error)
570  {
571  errno = cache_entry->m_cache_info.m_errno ? cache_entry->m_cache_info.m_errno : EIO;
572  throw false;
573  }
574 
575  errno = 0;
576  }
577  catch (bool _success)
578  {
579  success = _success;
580  }
581 
582  *bytes_read = static_cast<int>(len);
583 
584  return success;
585 }

587 bool transcoder_read_frame(Cache_Entry* cache_entry, char* buff, size_t offset, size_t len, uint32_t frame_no, int * bytes_read, LPVIRTUALFILE virtualfile)
588 {
589  bool success = false;
590 
591  Logging::trace(cache_entry->destname(), "Reading %1 bytes from offset %2 for frame no. %3.", len, offset, frame_no);
592 
593  // Store access time
594  cache_entry->update_access();
595 
596  // Update read counter
597  cache_entry->update_read_count();
598 
599  try
600  {
601  // Open for reading if necessary
602  if (!cache_entry->m_buffer->open_file(0, CACHE_FLAG_RO))
603  {
604  throw false;
605  }
606 
607  std::vector<uint8_t> data;
608 
609  // Wait until decoder thread has the requested frame available
610  if (cache_entry->m_is_decoding)
611  {
612  bool reported = false;
613 
614  // Try to read requested frame, stack a seek to if if this fails.
615  if (!cache_entry->m_buffer->read_frame(&data, frame_no))
616  {
617  cache_entry->m_seek_to_no = frame_no;
618  }
619 
620  int retries = FRAME_TIMEOUT * 1000 / GRANULARITY;
621  while (!cache_entry->m_buffer->read_frame(&data, frame_no) && !thread_exit)
622  {
623  if (errno != EAGAIN)
624  {
625  Logging::error(cache_entry->destname(), "Reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
626  throw false;
627  }
628 
629  if (errno == EAGAIN && !--retries)
630  {
631  errno = ETIMEDOUT;
632  Logging::error(cache_entry->destname(), "Timeout reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
633  throw false;
634  }
635 
636  if (fuse_interrupted())
637  {
638  Logging::info(cache_entry->destname(), "Client has gone away.");
639  throw false;
640  }
641 
642  if (thread_exit)
643  {
644  Logging::warning(cache_entry->destname(), "Received thread exit.");
645  throw false;
646  }
647 
648  if (!reported)
649  {
650  Logging::trace(cache_entry->destname(), "Cache miss at offset %<%11zu>1 (length %<%6u>2).", offset, len);
651  reported = true;
652  }
653 
654  const struct timespec ts = { 0, GRANULARITY MS };
655  nanosleep(&ts, nullptr);
656  }
657 
658  if (reported)
659  {
660  Logging::trace(cache_entry->destname(), "Frame no. %1: Cache hit at offset %<%11zu>2 (length %<%6u>3).", frame_no, offset, len);
661  }
662  success = !cache_entry->m_cache_info.m_error;
663  }
664  else
665  {
666  success = cache_entry->m_buffer->read_frame(&data, frame_no);
667  if (!success)
668  {
669  Logging::error(cache_entry->destname(), "Reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
670  throw false;
671  }
672  }
673 
674  if (success)
675  {
676  if (data.size() < offset)
677  {
678  len = 0;
679  }
680  else if (data.size() < offset + len)
681  {
682  len = data.size() - offset;
683  }
684 
685  if (len)
686  {
687  memcpy(buff, data.data() + offset, len);
688  }
689 
690  stat_set_size(&virtualfile->m_st, data.size());
691 
692  *bytes_read = static_cast<int>(len);
693  }
694  }
695  catch (bool _success)
696  {
697  success = _success;
698  }
699  return success;
700 }

755 static void transcoder_thread(void *arg)
756 {
757  THREAD_DATA *thread_data = static_cast<THREAD_DATA*>(arg);
758  Cache_Entry *cache_entry = static_cast<Cache_Entry *>(thread_data->m_arg);
759  FFmpeg_Transcoder transcoder;
760  int averror = 0;
761  int syserror = 0;
762  bool timeout = false;
763  bool success = true;
764 
765  std::unique_lock<std::recursive_mutex> lock(cache_entry->m_active_mutex);
766 
767  // Clear cache to remove any older remains
768  cache_entry->clear();
769 
770  // Must decode the file, otherwise simply use cache
771  cache_entry->m_is_decoding = true;
772 
773  try
774  {
775  bool unlocked = false;
776 
777  Logging::info(cache_entry->filename(), "Transcoding to %1.", params.current_format(cache_entry->virtualfile())->desttype().c_str());
778 
779  if (!cache_entry->open())
780  {
781  throw (static_cast<int>(errno));
782  }
783 
784  averror = transcoder.open_input_file(cache_entry->virtualfile());
785  if (averror < 0)
786  {
787  throw (static_cast<int>(errno));
788  }
789 
790  if (!cache_entry->m_cache_info.m_duration)
791  {
792  cache_entry->m_cache_info.m_duration = transcoder.duration();
793  }
794 
795  if (!cache_entry->m_cache_info.m_predicted_filesize)
796  {
797  cache_entry->m_cache_info.m_predicted_filesize = transcoder.predicted_filesize();
798  }
799 
800  if (!cache_entry->m_cache_info.m_video_frame_count)
801  {
802  cache_entry->m_cache_info.m_video_frame_count = transcoder.video_frame_count();
803  }
804 
805  if (!cache_entry->m_cache_info.m_segment_count)
806  {
807  cache_entry->m_cache_info.m_segment_count = transcoder.segment_count();
808  }
809 
810  if (!cache_entry->m_cache_info.m_duration)
811  {
812  cache_entry->m_cache_info.m_duration = transcoder.duration();
813  }
814 
815  if (!cache->maintenance(transcoder.predicted_filesize()))
816  {
817  throw (static_cast<int>(errno));
818  }
819 
820  averror = transcoder.open_output_file(cache_entry->m_buffer);
821  if (averror < 0)
822  {
823  throw (static_cast<int>(errno));
824  }
825 
826  memcpy(&cache_entry->m_id3v1, transcoder.id3v1tag(), sizeof(ID3v1));
827 
828  thread_data->m_initialised = true;
829 
830  unlocked = false;
831  if (!params.m_prebuffer_size || transcoder.is_frameset())
832  {
833  // Unlock frame set from beginning
834  unlocked = true;
835  thread_data->m_lock_guard = true;
836  thread_data->m_cond.notify_all(); // signal that we are running
837  }
838  else
839  {
840  Logging::debug(cache_entry->destname(), "Pre-buffering up to %1 bytes.", params.m_prebuffer_size);
841  }
842 
843  while (!cache_entry->is_finished() && !(timeout = cache_entry->decode_timeout()) && !thread_exit)
844  {
845  int status = 0;
846 
847  if (cache_entry->ref_count() > 1)
848  {
849  // Set last access time
850  cache_entry->update_access(false);
851  }
852 
853  if (transcoder.is_frameset())
854  {
855  uint32_t frame_no = cache_entry->m_seek_to_no;
856  if (frame_no)
857  {
858  cache_entry->m_seek_to_no = 0;
859 
860  averror = transcoder.stack_seek_frame(frame_no);
861  if (averror < 0)
862  {
863  throw (static_cast<int>(errno));
864  }
865  }
866  }
867  else if (transcoder.is_hls())
868  {
869  uint32_t segment_no = cache_entry->m_seek_to_no;
870  if (segment_no)
871  {
872  cache_entry->m_seek_to_no = 0;
873 
874  averror = transcoder.stack_seek_segment(segment_no);
875  if (averror < 0)
876  {
877  throw (static_cast<int>(errno));
878  }
879  }
880  }
881 
882  averror = transcoder.process_single_fr(status);
883  if (status < 0)
884  {
885  errno = EIO;
886  throw (static_cast<int>(errno));
887  }
888 
889  if (status == 1 && ((averror = transcode_finish(cache_entry, transcoder)) < 0))
890  {
891  errno = EIO;
892  throw (static_cast<int>(errno));
893  }
894 
895  if (!unlocked && cache_entry->m_buffer->buffer_watermark() > params.m_prebuffer_size)
896  {
897  unlocked = true;
898  Logging::debug(cache_entry->destname(), "Pre-buffer limit reached.");
899  thread_data->m_lock_guard = true;
900  thread_data->m_cond.notify_all(); // signal that we are running
901  }
902 
903  if (cache_entry->ref_count() <= 1 && cache_entry->suspend_timeout())
904  {
905  if (!unlocked && params.m_prebuffer_size)
906  {
907  unlocked = true;
908  thread_data->m_lock_guard = true;
909  thread_data->m_cond.notify_all(); // signal that we are running
910  }
911 
912  Logging::info(cache_entry->destname(), "Suspend timeout. Transcoding suspended after %1 seconds inactivity.", params.m_max_inactive_suspend);
913 
914  while (cache_entry->suspend_timeout() && !(timeout = cache_entry->decode_timeout()) && !thread_exit)
915  {
916  const struct timespec ts = { 0, GRANULARITY MS };
917  nanosleep(&ts, nullptr);
918  }
919 
920  if (timeout)
921  {
922  break;
923  }
924 
925  Logging::info(cache_entry->destname(), "Transcoding resumed.");
926  }
927  }
928 
929  if (!unlocked && params.m_prebuffer_size)
930  {
931  Logging::debug(cache_entry->destname(), "File transcode complete, releasing buffer early: Size %1.", cache_entry->m_buffer->buffer_watermark());
932  thread_data->m_lock_guard = true;
933  thread_data->m_cond.notify_all(); // signal that we are running
934  }
935  }
936  catch (int _syserror)
937  {
938  success = false;
939  syserror = _syserror;
940 
941  if (!syserror && averror > -512)
942  {
943  // If no system error reported explicitly, and averror is a POSIX error
944  // (we simply assume that if averror < 512, I think averrors are all higher values)
945  syserror = AVUNERROR(averror);
946  }
947 
948  cache_entry->m_is_decoding = false;
949  cache_entry->m_cache_info.m_error = true;
950  cache_entry->m_cache_info.m_errno = syserror ? syserror : EIO; // Preserve errno
951  cache_entry->m_cache_info.m_averror = averror; // Preserve averror
952 
953  thread_data->m_lock_guard = true;
954  thread_data->m_cond.notify_all(); // unlock main thread
955  }
956 
957  transcoder.close();
958 
959  if (timeout || thread_exit || transcoder.have_seeked())
960  {
961  cache_entry->m_is_decoding = false;
962 
963  if (!transcoder.have_seeked())
964  {
965  //cache_entry->m_cache_info.m_finished = RESULTCODE_FINISHED_ERROR;
966  cache_entry->m_cache_info.m_error = true;
967  cache_entry->m_cache_info.m_errno = EIO; // Report I/O error
968  cache_entry->m_cache_info.m_averror = averror; // Preserve averror
969 
970  if (timeout)
971  {
972  Logging::warning(cache_entry->destname(), "Timeout! Transcoding aborted after %1 seconds inactivity.", params.m_max_inactive_abort);
973  }
974  else
975  {
976  Logging::info(cache_entry->destname(), "Thread exit! Transcoding aborted.");
977  }
978  }
979  else
980  {
981  // Must restart from scratch, but this is not an error.
982  //cache_entry->m_cache_info.m_finished = RESULTCODE_FINISHED_INCOMPLETE;
983  cache_entry->m_cache_info.m_error = false;
984  cache_entry->m_cache_info.m_errno = 0;
985  cache_entry->m_cache_info.m_averror = 0;
986  }
987  }
988  else
989  {
990  cache_entry->m_is_decoding = false;
991  cache_entry->m_cache_info.m_error = !success;
992  cache_entry->m_cache_info.m_errno = success ? 0 : (syserror ? syserror : EIO); // Preserve errno
993  cache_entry->m_cache_info.m_averror = success ? 0 : averror; // Preserve averror
994 
995  if (success)
996  {
997  Logging::info(cache_entry->destname(), "Transcoding completed successfully.");
998  }
999  else
1000  {
1001  Logging::error(cache_entry->destname(), "Transcoding exited with error.");
1002  if (cache_entry->m_cache_info.m_errno)
1003  {
1004  Logging::error(cache_entry->destname(), "System error: (%1) %2", cache_entry->m_cache_info.m_errno, strerror(cache_entry->m_cache_info.m_errno));
1005  }
1006  if (cache_entry->m_cache_info.m_averror)
1007  {
1008  Logging::error(cache_entry->destname(), "FFMpeg error: (%1) %2", cache_entry->m_cache_info.m_averror, ffmpeg_geterror(cache_entry->m_cache_info.m_averror).c_str());
1009  }
1010  }
1011  }
1012 
1013  cache->close(&cache_entry, timeout ? CACHE_CLOSE_DELETE : CACHE_CLOSE_NOOPT);
1014 
1015  delete thread_data;
1016 
1017  errno = syserror;
1018 }
