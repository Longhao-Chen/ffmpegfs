90 int FFmpeg_Base::open_codec_context(AVCodecContext **avctx, int stream_idx, AVFormatContext *fmt_ctx, AVMediaType type, const char *filename) const
91 {
92  AVCodecContext *dec_ctx = nullptr;
93  AVCodec *dec = nullptr;
94  AVDictionary *opts = nullptr;
95  AVStream *input_stream;
96  AVCodecID codec_id = AV_CODEC_ID_NONE;
97  int ret;
98 
99  input_stream = fmt_ctx->streams[stream_idx];
100 
101  // Init the decoders, with or without reference counting
102  // av_dict_set_with_check(&opts, "refcounted_frames", refcount ? "1" : "0", 0);
103 
104 #if LAVF_DEP_AVSTREAM_CODEC
105  // allocate a new decoding context
106  dec_ctx = avcodec_alloc_context3(nullptr);
107  if (dec_ctx == nullptr)
108  {
109  Logging::error(filename, "Could not allocate a decoding context.");
110  return AVERROR(ENOMEM);
111  }
112 
113  // initialise the stream parameters with demuxer information
114  ret = avcodec_parameters_to_context(dec_ctx, input_stream->codecpar);
115  if (ret < 0)
116  {
117  return ret;
118  }
119 
120  codec_id = input_stream->codecpar->codec_id;
121 #else
122  dec_ctx = input_stream->codec;
123 
124  codec_id = dec_ctx->codec_id;
125 #endif
126 
127  // Find a decoder for the stream.
128  dec = avcodec_find_decoder(codec_id);
129  if (dec == nullptr)
130  {
131  Logging::error(filename, "Failed to find %1 input codec.", get_media_type_string(type));
132  return AVERROR(EINVAL);
133  }
134 
135  dec_ctx->codec_id = dec->id;
136 
137  ret = avcodec_open2(dec_ctx, dec, &opts);
138 
139  av_dict_free(&opts);
140 
141  if (ret < 0)
142  {
143  Logging::error(filename, "Failed to open %1 input codec for stream #%1 (error '%2').", get_media_type_string(type), input_stream->index, ffmpeg_geterror(ret).c_str());
144  return ret;
145  }
146 
147  Logging::debug(filename, "Opened input codec for stream #%1: %2", input_stream->index, get_codec_name(codec_id, true));
148 
149  *avctx = dec_ctx;
150 
151  return 0;
152 }

173 void FFmpeg_Base::video_stream_setup(AVCodecContext *output_codec_ctx, AVStream* output_stream, AVCodecContext *input_codec_ctx, AVRational framerate) const
174 {
175  AVRational time_base_tbn;
176  AVRational time_base_tbc;
177 
178  if (!framerate.num || !framerate.den)
179  {
180  framerate.num = 25;
181  framerate.den = 1;
182  Logging::warning(nullptr, "No information about the input framerate is available. Falling back to a default value of 25fps for output stream.");
183  }
184 
185  // timebase: This is the fundamental unit of time (in seconds) in terms
186  // of which frame timestamps are represented. For fixed-fps content,
187  // timebase should be 1/framerate and timestamp increments should be
188  // identical to 1.
189  //time_base = m_in.m_pVideo_stream->time_base;
190 
191  // tbn: must be set differently for the target format. Otherwise produces strange results.
192  switch (output_codec_ctx->codec_id)
193  {
194  case AV_CODEC_ID_THEORA: // ogg
195  case AV_CODEC_ID_MPEG1VIDEO:
196  case AV_CODEC_ID_MPEG2VIDEO:
197  {
198  time_base_tbn = av_inv_q(framerate);
199  time_base_tbc = time_base_tbn;
200  break;
201  }
202  case AV_CODEC_ID_VP9: // webm
203  {
204  time_base_tbn.num = 1;
205  time_base_tbn.den = 1000;
206  time_base_tbc = time_base_tbn;
207  break;
208  }
209  case AV_CODEC_ID_H264: // h264
210  {
211  time_base_tbn.num = 1;
212  time_base_tbn.den = 90000;
213  time_base_tbc = av_inv_q(framerate);
214  break;
215  }
216  default: // mp4 and all others
217  {
218  time_base_tbn.num = 1;
219  time_base_tbn.den = 90000;
220  time_base_tbc = time_base_tbn;
221  break;
222  }
223  }
224 
225  // tbn
226  output_stream->time_base = time_base_tbn;
227  // tbc
228  output_codec_ctx->time_base = time_base_tbc;
229 
230 #if !LAVF_DEP_AVSTREAM_CODEC
231  output_stream->codec->time_base = time_base_tbc;
232 #endif
233 
234  // tbr
235  // output_stream->r_frame_rate = m_in.m_pVideo_stream->r_frame_rate;
236  output_stream->r_frame_rate = framerate;
237 
238  // fps
239  output_stream->avg_frame_rate = framerate;
240  // output_codec_ctx->framerate = framerate;
241 
242  int alpha = 0;
243  int loss = 0;
244 
245  AVPixelFormat src_pix_fmt = input_codec_ctx->pix_fmt;
246  AVPixelFormat dst_pix_fmt = AV_PIX_FMT_NONE;
247  if (output_codec_ctx->codec->pix_fmts != nullptr)
248  {
249  dst_pix_fmt = avcodec_find_best_pix_fmt_of_list(output_codec_ctx->codec->pix_fmts, src_pix_fmt, alpha, &loss);
250  }
251 
252  if (dst_pix_fmt == AV_PIX_FMT_NONE)
253  {
254  // Fail safe if avcodec_find_best_pix_fmt_of_list has no idea what to use.
255  switch (output_codec_ctx->codec_id)
256  {
257  case AV_CODEC_ID_PRORES: // mov/prores
258  {
259  // yuva444p10le
260  // ProRes 4:4:4 if the source is RGB and ProRes 4:2:2 if the source is YUV.
261  dst_pix_fmt = AV_PIX_FMT_YUV422P10LE;
262  break;
263  }
264  default: // all others
265  {
266  // At this moment the output format must be AV_PIX_FMT_YUV420P;
267  dst_pix_fmt = AV_PIX_FMT_YUV420P;
268  break;
269  }
270  }
271  }
272 
273  output_codec_ctx->pix_fmt = dst_pix_fmt;
274  output_codec_ctx->gop_size = 12; // emit one intra frame every twelve frames at most
275 }

301 void FFmpeg_Base::video_info(bool out_file, const AVFormatContext *format_ctx, const AVStream *stream) const
302 {
303  int64_t duration = AV_NOPTS_VALUE;
304 
305  if (stream->duration != AV_NOPTS_VALUE)
306  {
307  duration = av_rescale_q_rnd(stream->duration, stream->time_base, av_get_time_base_q(), static_cast<AVRounding>(AV_ROUND_UP | AV_ROUND_PASS_MINMAX));
308  }
309 
310  Logging::debug(out_file ? destname() : filename(), "Video %1: %2@%3 [%4]",
311  out_file ? "out" : "in",
312  get_codec_name(CODECPAR(stream)->codec_id, false),
313  format_bitrate((CODECPAR(stream)->bit_rate != 0) ? CODECPAR(stream)->bit_rate : format_ctx->bit_rate).c_str(),
314  format_duration(duration).c_str());

353 uint32_t FFmpeg_Base::pts_to_frame(AVStream* stream, int64_t pts) const
354 {
355  if (pts == AV_NOPTS_VALUE)
356  {
357  return 0;
358  }
359  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
360  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
361  return static_cast<uint32_t>(av_rescale(pts - start_time, factor.num, factor.den) + 1);
362 }

364 int64_t FFmpeg_Base::frame_to_pts(AVStream* stream, uint32_t frame_no) const
365 {
366  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
367  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
368  return static_cast<uint32_t>(av_rescale(frame_no - 1, factor.den, factor.num) + start_time);
