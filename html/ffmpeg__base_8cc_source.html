83 void FFmpeg_Base::video_stream_setup(AVCodecContext *output_codec_ctx, AVStream* output_stream, AVCodecContext *input_codec_ctx, AVRational framerate, AVPixelFormat enc_hw_pix_fmt) const
84 {
85  AVRational time_base_tbn;
86  AVRational time_base_tbc;
87 
88  if (!framerate.num || !framerate.den)
89  {
90  framerate.num = 25;
91  framerate.den = 1;
92  Logging::warning(nullptr, "No information about the input framerate is available. Falling back to a default value of 25fps for output stream.");
93  }
94 
95  // timebase: This is the fundamental unit of time (in seconds) in terms
96  // of which frame timestamps are represented. For fixed-fps content,
97  // timebase should be 1/framerate and timestamp increments should be
98  // identical to 1.
99  //time_base = m_in.m_pVideo_stream->time_base;
100 
101  // tbn: must be set differently for the target format. Otherwise produces strange results.
102  switch (output_codec_ctx->codec_id)
103  {
104  case AV_CODEC_ID_THEORA: // ogg
105  case AV_CODEC_ID_MPEG1VIDEO:
106  case AV_CODEC_ID_MPEG2VIDEO:
107  {
108  time_base_tbn = av_inv_q(framerate);
109  time_base_tbc = time_base_tbn;
110  break;
111  }
112  case AV_CODEC_ID_VP9: // webm
113  {
114  time_base_tbn.num = 1;
115  time_base_tbn.den = 1000;
116  time_base_tbc = time_base_tbn;
117  break;
118  }
119  case AV_CODEC_ID_H264: // h264
120  case AV_CODEC_ID_H265: // h265
121  {
122  time_base_tbn.num = 1;
123  time_base_tbn.den = 90000;
124  time_base_tbc = av_inv_q(framerate);
125  break;
126  }
127  default: // mp4 and all others
128  {
129  time_base_tbn.num = 1;
130  time_base_tbn.den = 90000;
131  time_base_tbc = time_base_tbn;
132  break;
133  }
134  }
135 
136  // tbn
137  output_stream->time_base = time_base_tbn;
138  // tbc
139  output_codec_ctx->time_base = time_base_tbc;
140 
141 #if !LAVF_DEP_AVSTREAM_CODEC
142  output_stream->codec->time_base = time_base_tbc;
143 #endif
144 
145  // tbr
146  // output_stream->r_frame_rate = m_in.m_pVideo_stream->r_frame_rate;
147  output_stream->r_frame_rate = framerate;
148 
149  // fps
150  output_stream->avg_frame_rate = framerate;
151  // output_codec_ctx->framerate = framerate;
152 
153  if (enc_hw_pix_fmt == AV_PIX_FMT_NONE)
154  {
155  // Automatic pix_fmt selection
156  int loss = 0;
157 
158  AVPixelFormat src_pix_fmt = input_codec_ctx->pix_fmt;
159  if (output_codec_ctx->codec->pix_fmts != nullptr)
160  {
161  int alpha = 0;
162  enc_hw_pix_fmt = avcodec_find_best_pix_fmt_of_list(output_codec_ctx->codec->pix_fmts, src_pix_fmt, alpha, &loss);
163  }
164 
165  if (enc_hw_pix_fmt == AV_PIX_FMT_NONE)
166  {
167  // Fail safe if avcodec_find_best_pix_fmt_of_list has no idea what to use.
168  switch (output_codec_ctx->codec_id)
169  {
170  case AV_CODEC_ID_PRORES: // mov/prores
171  {
172  // yuva444p10le
173  // ProRes 4:4:4 if the source is RGB and ProRes 4:2:2 if the source is YUV.
174  enc_hw_pix_fmt = AV_PIX_FMT_YUV422P10LE;
175  break;
176  }
177  default: // all others
178  {
179  // At this moment the output format must be AV_PIX_FMT_YUV420P;
180  enc_hw_pix_fmt = AV_PIX_FMT_YUV420P;
181  break;
182  }
183  }
184  }
185  }
186 
187  output_codec_ctx->pix_fmt = enc_hw_pix_fmt;
188  output_codec_ctx->gop_size = 12; // emit one intra frame every twelve frames at most
189 }

237 void FFmpeg_Base::video_info(bool out_file, const AVFormatContext *format_ctx, const AVStream *stream) const
238 {
239  int64_t duration = AV_NOPTS_VALUE;
240 
241  if (stream->duration != AV_NOPTS_VALUE)
242  {
243  duration = av_rescale_q_rnd(stream->duration, stream->time_base, av_get_time_base_q(), static_cast<AVRounding>(AV_ROUND_UP | AV_ROUND_PASS_MINMAX));
244  }
245 
246  Logging::debug(out_file ? destname() : filename(), "Video %1: %2@%3 [%4]",
247  out_file ? "out" : "in",
248  get_codec_name(CODECPAR(stream)->codec_id, false),
249  format_bitrate((CODECPAR(stream)->bit_rate != 0) ? CODECPAR(stream)->bit_rate : format_ctx->bit_rate).c_str(),
250  format_duration(duration).c_str());
251 }

289 uint32_t FFmpeg_Base::pts_to_frame(AVStream* stream, int64_t pts) const
290 {
291  if (pts == AV_NOPTS_VALUE)
292  {
293  return 0;
294  }
295  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
296  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
297  return static_cast<uint32_t>(av_rescale(pts - start_time, factor.num, factor.den) + 1);
298 }

300 int64_t FFmpeg_Base::frame_to_pts(AVStream* stream, uint32_t frame_no) const
301 {
302  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
303  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
304  return static_cast<uint32_t>(av_rescale(frame_no - 1, factor.den, factor.num) + start_time);
305 }
