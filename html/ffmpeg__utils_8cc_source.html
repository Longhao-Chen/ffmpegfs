558 int show_caps(int device_only)
559 {
560  const AVInputFormat *ifmt = nullptr;
561  const AVOutputFormat *ofmt = nullptr;
562  const char *last_name;
563  int is_dev;
564 
565  std::printf("%s\n"
566  " D. = Demuxing supported\n"
567  " .E = Muxing supported\n"
568  " --\n", device_only ? "Devices:" : "File formats:");
569  last_name = "000";
570  for (;;)
571  {
572  int decode = 0;
573  int encode = 0;
574  const char *name = nullptr;
575  const char *long_name = nullptr;
576  const char *extensions = nullptr;
577 
578 #if LAVF_DEP_AV_REGISTER
579  void *ofmt_opaque = nullptr;
580  ofmt_opaque = nullptr;
581  while ((ofmt = av_muxer_iterate(&ofmt_opaque)))
582 #else
583  while ((ofmt = av_oformat_next(ofmt)))
584 #endif
585  {
586  is_dev = is_device(ofmt->priv_class);
587  if (!is_dev && device_only)
588  {
589  continue;
590  }
591 
592  if ((!name || strcmp(ofmt->name, name) < 0) &&
593  strcmp(ofmt->name, last_name) > 0)
594  {
595  name = ofmt->name;
596  long_name = ofmt->long_name;
597  encode = 1;
598  }
599  }
600 #if LAVF_DEP_AV_REGISTER
601  void *ifmt_opaque = nullptr;
602  ifmt_opaque = nullptr;
603  while ((ifmt = av_demuxer_iterate(&ifmt_opaque)))
604 #else
605  while ((ifmt = av_iformat_next(ifmt)))
606 #endif
607  {
608  is_dev = is_device(ifmt->priv_class);
609  if (!is_dev && device_only)
610  {
611  continue;
612  }
613 
614  if ((!name || strcmp(ifmt->name, name) < 0) &&
615  strcmp(ifmt->name, last_name) > 0)
616  {
617  name = ifmt->name;
618  long_name = ifmt->long_name;
619  extensions = ifmt->extensions;
620  encode = 0;
621  }
622  if (name && strcmp(ifmt->name, name) == 0)
623  {
624  decode = 1;
625  }
626  }
627 
628  if (!name)
629  {
630  break;
631  }
632 
633  last_name = name;
634  if (!extensions)
635  {
636  continue;
637  }
638 
639  std::printf(" %s%s %-15s %-15s %s\n",
640  decode ? "D" : " ",
641  encode ? "E" : " ",
642  extensions != nullptr ? extensions : "-",
643  name,
644  long_name ? long_name:" ");
645  }
646  return 0;
647 }

673 {
674  char *buffer = new_strdup(path);
675 
676  if (buffer == nullptr)
677  {
678  return ENOMEM;
679  }
680 
681  std::string dir;
682  char *p = strtok (buffer, "/");
683  int status = 0;
684 
685  while (p != nullptr)
686  {
687  int newstat;
688 
689  dir += "/";
690  dir += p;
691 
692  errno = 0;
693 
694  newstat = mkdir(dir.c_str(), mode);
695 
696  if (!status && newstat && errno != EEXIST)
697  {
698  status = -1;
699  break;
700  }
701 
702  status = newstat;
703 
704  p = strtok (nullptr, "/");
705  }
706 
707  delete [] buffer;
708 
709  return status;
710 }

1164 int is_mount(const std::string & path)
1165 {
1166  char* orig_name = nullptr;
1167  int ret = 0;
1168 
1169  try
1170  {
1171  struct stat file_stat;
1172  struct stat parent_stat;
1173  char * parent_name = nullptr;
1174 
1175  orig_name = new_strdup(path);
1176 
1177  if (orig_name == nullptr)
1178  {
1179  std::fprintf(stderr, "is_mount(): Out of memory\n");
1180  errno = ENOMEM;
1181  throw -1;
1182  }
1183 
1184  // get the parent directory of the file
1185  parent_name = dirname(orig_name);
1186 
1187  // get the file's stat info
1188  if (-1 == stat(path.c_str(), &file_stat))
1189  {
1190  std::fprintf(stderr, "is_mount(): (%i) %s\n", errno, strerror(errno));
1191  throw -1;
1192  }
1193 
1194  //determine whether the supplied file is a directory
1195  // if it isn't, then it can't be a mountpoint.
1196  if (!(file_stat.st_mode & S_IFDIR))
1197  {
1198  std::fprintf(stderr, "is_mount(): %s is not a directory.\n", path.c_str());
1199  throw -1;
1200  }
1201 
1202  // get the parent's stat info
1203  if (-1 == stat(parent_name, &parent_stat))
1204  {
1205  std::fprintf(stderr, "is_mount(): (%i) %s\n", errno, strerror(errno));
1206  throw -1;
1207  }
1208 
1209  // if file and parent have different device ids,
1210  // then the file is a mount point
1211  // or, if they refer to the same file,
1212  // then it's probably the root directory
1213  // and therefore a mountpoint
1214  if (file_stat.st_dev != parent_stat.st_dev ||
1215  (file_stat.st_dev == parent_stat.st_dev &&
1216  file_stat.st_ino == parent_stat.st_ino))
1217  {
1218  // IS a mountpoint
1219  ret = 1;
1220  }
1221  else
1222  {
1223  // is NOT a mountpoint
1224  ret = 0;
1225  }
1226  }
1227  catch (int _ret)

1229  ret = _ret;
1230  }

1233
