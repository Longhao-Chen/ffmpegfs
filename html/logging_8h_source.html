303  template <typename T, typename... Args>
304  static std::string format_helper(
305  const std::string &string_to_search,
306  const size_t index_to_replace,
307  T &&val,
308  Args &&...args)
309  {
310  // Match %# exactly (e.g. %12 and %123 literally)
311 
312  std::regex exp("%(<([^>]+)>)*" + std::to_string(index_to_replace) + "(?=[^0-9]|$)");
313  std::smatch res;
314  std::string string_to_update(string_to_search);
315  std::string::const_iterator searchStart(string_to_search.cbegin());
316  size_t offset = 0;
317 
318  while (std::regex_search(searchStart, string_to_search.cend(), res, exp))
319  {
320  std::ostringstream ostr;
321 
322  if (res[2].length())
323  {
324  // Found match with printf format in res[2]
325  size_t size = static_cast<size_t>(std::snprintf(nullptr, 0, res[2].str().c_str(), val)) + 1;
326  std::vector<char> buffer;
327  buffer.resize(size);
328  std::snprintf(buffer.data(), size, res[2].str().c_str(), val);
329  ostr << buffer.data();
330  }
331  else
332  {
333  // No printf format, replace literally
334  ostr << val;
335  }
336 
337  string_to_update.replace(static_cast<size_t>(res.position()) + offset, static_cast<size_t>(res[0].length()), ostr.str());
338 
339  offset += static_cast<size_t>(res.position()) + ostr.str().length();
340 
341  searchStart = res.suffix().first;
342  }
343 
344  return format_helper(
345  string_to_update,
346  index_to_replace + 1,
347  std::forward<Args>(args)...);
348  }
