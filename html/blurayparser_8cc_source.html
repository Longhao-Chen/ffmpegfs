327 static bool create_bluray_virtualfile(BLURAY *bd, const BLURAY_TITLE_INFO* ti, const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler, bool is_main_title, bool full_title, uint32_t title_idx, uint32_t chapter_idx)
328 {
329  BLURAY_CLIP_INFO *clip = &ti->clips[0];
330  BLURAY_TITLE_CHAPTER *chapter = &ti->chapters[chapter_idx];
331  char title_buf[PATH_MAX + 1];
332  struct stat stbuf;
333  int64_t duration;
334 
335  if (full_title)
336  {
337  duration = static_cast<int64_t>(ti->duration) * AV_TIME_BASE / 90000;
338 
339  if (duration < AV_TIME_BASE)
340  {
341  Logging::debug(path, "Title %1: skipping empty title", title_idx + 1);
342  return true;
343  }
344 
345  snprintf(title_buf, sizeof(title_buf) - 1, "%02d. Title [%s]%s.%s",
346  title_idx + 1,
347  replace_all(format_duration(duration), ":", "-").c_str(),
348  is_main_title ? "+" : "",
349  params.m_format[0].fileext().c_str()); // can safely assume this is a video format
350  }
351  else
352  {
353  duration = static_cast<int64_t>(chapter->duration) * AV_TIME_BASE / 90000;
354 
355  if (duration < AV_TIME_BASE)
356  {
357  Logging::debug(path, "Title %1 Chapter %2: skipping empty chapter", title_idx + 1, chapter_idx + 1);
358  return true;
359  }
360 
361  snprintf(title_buf, sizeof(title_buf) - 1, "%02d. Chapter %03d [%s]%s.%s",
362  title_idx + 1,
363  chapter_idx + 1,
364  replace_all(format_duration(duration), ":", "-").c_str(),
365  is_main_title ? "+" : "",
366  params.m_format[0].fileext().c_str()); // can safely assume this is a video format
367 
368  }
369 
370  std::string filename(title_buf);
371 
372  memcpy(&stbuf, statbuf, sizeof(struct stat));
373 
374 #if defined __x86_64__ || !defined __USE_FILE_OFFSET64
375  stbuf.st_size = 0; //static_cast<__off_t>(size);
376 #else
377  stbuf.st_size = 0; //static_cast<__off64_t>(size);
378 #endif
379  stbuf.st_blocks = (stbuf.st_size + 512 - 1) / 512;
380 
381  if (buf != nullptr && filler(buf, title_buf, &stbuf, 0))
382  {
383  // break;
384  }
385 
386  LPVIRTUALFILE virtualfile = nullptr;
387  if (!params.m_format[0].is_multiformat())
388  {
389  virtualfile = insert_file(VIRTUALTYPE_BLURAY, path + filename, &stbuf);
390  }
391  else
392  {
393  std::string origpath(path + filename);
394 
395  int flags = VIRTUALFLAG_DIRECTORY | VIRTUALFLAG_FILESET;
396 
397  // Change file to directory for the frame set
398  // Change file to virtual directory for the frame set. Keep permissions.
399  stbuf.st_mode &= ~static_cast<mode_t>(S_IFREG | S_IFLNK);
400  stbuf.st_mode |= S_IFDIR;
401  stbuf.st_nlink = 2;
402  stbuf.st_size = stbuf.st_blksize;
403 
404  append_sep(&origpath);
405 
406  if (params.m_format[0].is_frameset())
407  {
408  flags |= VIRTUALFLAG_FRAME;
409  }
410  else if (params.m_format[0].is_hls())
411  {
412  flags |= VIRTUALFLAG_HLS;
413  }
414 
415  virtualfile = insert_file(VIRTUALTYPE_BLURAY, origpath, &stbuf, flags);
416  }
417 
418  // Bluray is video format anyway
419  virtualfile->m_format_idx = 0;
420  // Mark title/chapter/angle
421  virtualfile->m_full_title = full_title;
422  virtualfile->m_bluray.m_title_no = title_idx + 1;
423  virtualfile->m_bluray.m_playlist_no = ti->playlist;
424  virtualfile->m_bluray.m_chapter_no = chapter_idx + 1;
425  virtualfile->m_bluray.m_angle_no = 1;
426 
427  if (!transcoder_cached_filesize(virtualfile, &stbuf))
428  {
429  BITRATE video_bit_rate = 29*1024*1024; // In case the real bitrate cannot be calculated later, assume 20 Mbit video bitrate
430  BITRATE audio_bit_rate = 256*1024; // In case the real bitrate cannot be calculated later, assume 256 kBit audio bitrate
431 
432  int channels = 0;
433  int sample_rate = 0;
434  int audio = 0;
435 
436  int width = 0;
437  int height = 0;
438  AVRational framerate = { 0, 0 };
439  int interleaved = 0;
440 
441  if (!bd_select_title(bd, title_idx))
442  {
443  Logging::error(path, "Failed to open bluray title %1", title_idx);
444  errno = EIO;
445  return false;
446  }
447 
448  uint64_t size = bd_get_title_size(bd);
449 
450  virtualfile->m_duration = duration;
451 
452  if (duration)
453  {
456  video_bit_rate = static_cast<BITRATE>(size * 8LL * AV_TIME_BASE / static_cast<uint64_t>(duration)); // calculate bitrate in bps
457  }
458 
459  // Get details
460  if (clip->audio_stream_count)
461  {
462  stream_info(path, &clip->audio_streams[parse_find_best_audio_stream()], &channels, &sample_rate, &audio, &width, &height, &framerate, &interleaved);
463  }
464  if (clip->video_stream_count)
465  {
466  stream_info(path, &clip->video_streams[parse_find_best_video_stream()], &channels, &sample_rate, &audio, &width, &height, &framerate, &interleaved);
467  }
468 
469  Logging::trace(virtualfile->m_origfile, "Video %1 %2x%3@%<%5.2f>4%5 fps %6 [%7]", format_bitrate(video_bit_rate).c_str(), width, height, av_q2d(framerate), interleaved ? "i" : "p", format_size(size).c_str(), format_duration(duration).c_str());
470  if (audio > -1)
471  {
472  Logging::trace(virtualfile->m_origfile, "Audio %1 channels %2", channels, format_samplerate(sample_rate).c_str());
473  }
474 
475  transcoder_set_filesize(virtualfile, duration, audio_bit_rate, channels, sample_rate, video_bit_rate, width, height, interleaved, framerate);
476 
477  virtualfile->m_video_frame_count = static_cast<uint32_t>(av_rescale_q(duration, av_get_time_base_q(), av_inv_q(framerate)));
478  virtualfile->m_predicted_size = size;
479  }
480 
481  return true;
482 }

492 static int parse_bluray(const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler)
493 {
494  BLURAY *bd;
495  uint32_t title_count;
496  int main_title;
497  unsigned int seconds = 0;
498  uint8_t flags = TITLES_RELEVANT;
499  const char *bd_dir = nullptr;
500  bool success = true;
501 
502  bd_dir = path.c_str();
503 
504  Logging::debug(path, "Parsing Bluray.");
505 
506  bd = bd_open(bd_dir, nullptr);
507 
508  title_count = bd_get_titles(bd, flags, seconds);
509  main_title = bd_get_main_title(bd);
510  if (main_title >= 0)
511  {
512  Logging::trace(path, "Main title: %1", main_title + 1);
513  }
514 
515  for (uint32_t title_idx = 0; title_idx < title_count && success; title_idx++)
516  {
517  BLURAY_TITLE_INFO* ti = bd_get_title_info(bd, title_idx, 0);
518  bool is_main_title = (main_title >= 0 && title_idx == static_cast<uint32_t>(main_title));
519 
520  // Add separate chapters
521  for (uint32_t chapter_idx = 0; chapter_idx < ti->chapter_count && success; chapter_idx++)
522  {
523  success = create_bluray_virtualfile(bd, ti, path, statbuf, buf, filler, is_main_title, false, title_idx, chapter_idx);
524  }
525 
526  if (success && ti->chapter_count > 1)
527  {
528  // If more than 1 chapter, add full title as well
529  success = create_bluray_virtualfile(bd, ti, path, statbuf, buf, filler, is_main_title, true, title_idx, 0);
530  }
531 
532  bd_free_title_info(ti);
533  }
534 
535  bd_close(bd);
536 
537  if (success)
538  {
539  return static_cast<int>(title_count);
540  }
541  else
542  {
543  return -errno;
544  }
545 }

548 {

552 
553  append_sep(&path);
554 
555  if (stat((path + "BDMV/index.bdmv").c_str(), &stbuf) == 0)
